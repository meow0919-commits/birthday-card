<script>

    function generateCard() {

        const name = document.getElementById('name').value;

        const gender = document.getElementById('gender').value;

        const hobby = document.getElementById('hobby').value;

        const resultDiv = document.getElementById('result');


        if(!name || !hobby) {

            alert("请填完所有信息哦！");

            return;

        }


        // 简单的逻辑生成模板

        const templates = [

            `亲爱的 ${name}${gender}：\n\n祝你生日快乐！听说你最近沉迷于${hobby}，希望你在新的一年里，能在这个领域里收获更多快乐与惊喜。愿你的生活像你的爱好一样多姿多彩！`,

            `致 ${name}：\n\n生日快乐！在这个特别的日子里，愿你在${hobby}的道路上越走越远，遇见更多志同道合的朋友。愿新的一岁，平安喜乐，万事胜意！`

        ];


        const randomText = templates[Math.floor(Math.random() * templates.length)];

        resultDiv.innerText = randomText;

        resultDiv.style.display = 'block';

    }

</script>
